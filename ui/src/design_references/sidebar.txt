<script>
      const sidebar = document.getElementById('sidebar');
      const collapseButton = document.getElementById('collapse-button');
      const mainContent = document.querySelector('.layout-content-container');
      let isManuallyCollapsed = true; // Start collapsed
      function setSidebarState(collapsed) {
          if (collapsed) {
              sidebar.classList.remove('sidebar-expanded');
              sidebar.classList.add('sidebar-collapsed');
              mainContent.style.marginLeft = '96px'; // sidebar width + right padding
              // Switch icons for collapse button
              collapseButton.querySelector('.collapse-icon-open').classList.remove('hidden');
              collapseButton.querySelector('.collapse-icon-open').classList.add('block');
              collapseButton.querySelector('.collapse-icon-close').classList.remove('block');
              collapseButton.querySelector('.collapse-icon-close').classList.add('hidden');
          } else {
              sidebar.classList.remove('sidebar-collapsed');
              sidebar.classList.add('sidebar-expanded');
              mainContent.style.marginLeft = '296px'; // sidebar width + right padding
              // Switch icons for collapse button
              collapseButton.querySelector('.collapse-icon-open').classList.remove('block');
              collapseButton.querySelector('.collapse-icon-open').classList.add('hidden');
              collapseButton.querySelector('.collapse-icon-close').classList.remove('hidden');
              collapseButton.querySelector('.collapse-icon-close').classList.add('block');
          }
      }
      collapseButton.addEventListener('click', () => {
          isManuallyCollapsed = !isManuallyCollapsed;
          setSidebarState(isManuallyCollapsed);
      });
      // Auto-expand on hover only if manually collapsed
      sidebar.addEventListener('mouseenter', () => {
        if(isManuallyCollapsed) {
          setSidebarState(false); 
        }
      });
      // Auto-collapse on mouseleave only if it was auto-expanded (i.e. isManuallyCollapsed is true)
      sidebar.addEventListener('mouseleave', () => {
        if(isManuallyCollapsed) {
          setSidebarState(true);
        }
      });
      // Initialize sidebar state
      setSidebarState(isManuallyCollapsed);
      // Adjust main content margin based on initial sidebar state
      // This avoids a jump if the sidebar starts expanded.
      if(sidebar.classList.contains('sidebar-expanded')) {
        mainContent.style.marginLeft = '296px';
      } else {
        mainContent.style.marginLeft = '96px';
      }
    </script>
